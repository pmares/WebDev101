Setting / Altering Element Attributes

setAttribute

demo.html
<button onclick="myFunction()"> Click Me </button>
demo.js
function myFunction() { let element = document.getElementById("demo-color"); if (element.className === 'color-change') { element.setAttribute("class",""); } else { element.setAttribute("class","color-change"); } }
style.css
#demo-color { background-color:lightgreen; } #demo-color.color-change { background-color:blue; }
