Scope

A variable is "scoped" to the block in which it is defined. Outsize of the block, the variable is undefined

var a = 0; function outer() { var a = 1; function inner() { var b = 2; // we can access both `a` and `b` here console.log( a + b ); // 3 } inner(); // we can only access `a` here console.log( a ); // 1 console.log( typeof(b) ); // "undefined" } outer(); console.log(a);
